public class LongestIncreasingSubsequence {
    public int lengthOfLIS(int[] nums) {  
        int[] dp = new int[nums.length];  
        Arrays.fill(dp, 1);  
        int ans = 1;  
  
        for (int i = 0; i < nums.length; i++) {  
            for (int j = 0; j < i; j++) {  
                if (nums[i] > nums[j] && dp[i] < dp[j] + 1) {  
                    dp[i] = dp[j] + 1;  
                    ans = Math.max(ans, dp[i]);  
                }            
            }        
        }        
        return ans;  
    }
}
